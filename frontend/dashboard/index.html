/** * Dashboard Controller * -------------------- * Connects the dashboard UI to the IAI engine. */ import bridge from "../bridge/bridge-connection.js"; class DashboardController { constructor() { this.statusEl = null; this.runBtn = null; this.inputEl = null; this.outputEl = null; } init() { this.statusEl = document.getElementById("engine-status"); this.runBtn = document.getElementById("run-btn"); this.inputEl = document.getElementById("dashboard-input"); this.outputEl = document.getElementById("dashboard-output"); // Connect to bridge bridge.connect(); // Listen for brain readiness bridge.on("brainReady", () => { this.updateStatus("🟢 Engine Ready"); }); // Listen for responses bridge.on("brainResponse", (result) => { this.outputEl.innerText = result.final; }); // Wire run button this.runBtn.addEventListener("click", () => { const prompt = this.inputEl.value.trim(); if (!prompt) return; bridge.sendPrompt(prompt); this.updateStatus("⚡ Processing..."); }); } updateStatus(text) { if (this.statusEl) this.statusEl.innerText = text; } } export default new DashboardController();
